CREATE TABLE Product(
	ID INT PRIMARY KEY IDENTITY(1,1),
	Name VARCHAR(50),
	Quantity BIGINT,
	Price BIGINT,
	AddDateTime DATETIME DEFAULT GETDATE(),
	AddedBy BIGINT,
	UpdateDateTime DATETIME,
	UpdatedBy BIGINT,
	IsActive BIT DEFAULT 1
);
